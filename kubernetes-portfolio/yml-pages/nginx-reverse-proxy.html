NGINX Revers Proxy Configuration


        apiVersion: apps/v1
        kind: Deployment
        metadata:
          name: nginx-reverse-proxy
          labels:
            app: nginx-reverse-proxy
        spec:
          replicas: 1
          selector:
            matchLabels:
              app: nginx-reverse-proxy
          template:
            metadata:
              labels:
                app: nginx-reverse-proxy
            spec:
              containers:
                - name: nginx
                  image: nginx
                  ports:
                    - containerPort: 80
                  volumeMounts:
                    - mountPath: /etc/nginx/nginx.conf
                      name: config-volume
              volumes:
                - name: config-volume
                  configMap:
                    name: nginx-config
